Course Schedule - LeetCode#

class Solution {
    public boolean canFinish(int numCourses, int[][] prerequisites) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < numCourses; i++) {
            map.put(i, new ArrayList<>());
        }
        for (int[] pre : prerequisites) {
            map.get(pre[0]).add(pre[1]);
        }
        int[] visited = new int[numCourses];
        for (int i = 0; i < numCourses; i++) {
            if (!dfs(i, map, visited)) {
                return false;
            }
        }
        return true;
    }

    private boolean dfs(int c, Map<Integer, List<Integer>> map, int[] visited) {
        if (visited[c] == 2) {
            return true;
        }
        visited[c] = 1;
        List<Integer> list = map.get(c);
        for (int i : list) {
            if (visited[i] == 1) {
                return false;
            }
            if (visited[i] == 0) {
                if (!dfs(i, map, visited)) {
                    return false;
                }
            }
        }
        visited[c] = 2;
        return true;
    }
    
    public static void main(String[] args) {
        Solution solution = new Solution();
        int numCourses = 4;
        int[][] prerequisites = {{1, 0}, {2, 0}, {3, 1}, {3, 2}};
        
        boolean canFinish = solution.canFinish(numCourses, prerequisites);
        System.out.println("Can finish all courses: " + canFinish);
    }
}

Course Schedule II - LeetCode#

import java.util.*;

class Solution {
    public int[] findOrder(int numCourses, int[][] prerequisites) {
        // visited 3种状态
        // 0：未访问
        // 1：正在访问
        // 2：访问完
        int[] visited = new int[numCourses];
        Map<Integer, List<Integer>> map = new HashMap<>();
        Deque<Integer> queue = new LinkedList<>();
        
        for (int i = 0; i < numCourses; i++) {
            map.put(i, new ArrayList<>());
        }
        for (int[] pre : prerequisites) {
            map.get(pre[0]).add(pre[1]);
        }
        // 以上是完成邻接表的过程，每个key的list储存的是他的pre课程
        for (int i = 0; i < numCourses; i++) {
            if (!dfs(i, visited, map, queue)) {
                return new int[0];
            }
        }
        int[] res = new int[numCourses];
        for (int i = 0; i < numCourses; i++) {
            res[i] = queue.poll();
        }
        return res;
    }

    private boolean dfs(int c, int[] visited, Map<Integer, List<Integer>> map, Deque<Integer> queue) {
        if (visited[c] == 2) {
            return true;
        }
        visited[c] = 1;
        for (int i : map.get(c)) {
            if (visited[i] == 1) {
                // 检测到环
                return false;
            }
            if (visited[i] == 0) {
                if (!dfs(i, visited, map, queue)) {
                    return false;
                }
            }
        }
        visited[c] = 2;
        queue.offer(c);
        return true;
    }

    static public void main(String[] args) {
        Solution cs = new Solution();
        int numCourses = 4;
        int[][] prerequisites = {{3,0},{0,1}};
        
        int[] order = cs.findOrder(numCourses, prerequisites);
        System.out.println(Arrays.toString(order)); // 输出拓扑排序结果
    }
}

判断是否为质数#

public class PrimeChecker {
    public static boolean isPrime(int num) {
        if (num <= 1) {
            return false; // 1和更小的数不是质数
        }
        if (num == 2) {
            return true; // 2是质数
        }
        if (num % 2 == 0) {
            return false; // 排除所有偶数
        }
        for (int i = 3; i <= Math.sqrt(num); i += 2) {
            if (num % i == 0) {
                return false; // 如果找到因数，则不是质数
            }
        }
        return true; // 如果没有找到因数，则是质数
    }

    public static void main(String[] args) {
        int num = 29; // 测试的数字
        System.out.println("Is " num + "prime number? " + isPrime(num));
    }
}

数组的最大公约数#

public class GCDArray {
    // 求两个数的最大公因数
    private static int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }

    // 求数组中所有数的最大公因数
    public static int findGCD(int[] nums) {
        if (nums == null || nums.length == 0) {
            // 如果数组为空则返回0
            return 0;
        }
        int result = nums[0];
        for (int i = 1; i < nums.length; i++) {
            result = gcd(result, nums[i]);
            // 如果结果为1，可以提前终止，因为1是最小的公因数
            if (result == 1) {
                return 1;
            }
        }
        return result;
    }

    public static void main(String[] args) {
        int[] nums = {24, 36, 48}; // 示例数组
        System.out.println("GCD of this array is: " + findGCD(nums));
    }
}

数组的最小公倍数#

public class LCMArray {
    // 求两个数的最大公因数
    private static int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }

    // 求两个数的最小公倍数
    private static int lcm(int a, int b) {
        return (a * b) / gcd(a, b);
    }

    // 求数组中所有数的最小公倍数
    public static int findLCM(int[] nums) {
        if (nums == null || nums.length == 0) {
            // 如果数组为空则返回0
            return 0;
        }
        int result = nums[0];
        for (int i = 1; i < nums.length; i++) {
            result = lcm(result, nums[i]);
        }
        return result;
    }

    public static void main(String[] args) {
        int[] nums = {4, 5, 6}; // 示例数组
        System.out.println("LCM if this array is: " + findLCM(nums));
    }
}
